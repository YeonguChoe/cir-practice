```cpp
bool IRTranslator::translateConstrainedFPIntrinsic(
    const ConstrainedFPIntrinsic &FPI, MachineIRBuilder &MIRBuilder) {
  fp::ExceptionBehavior EB = *FPI.getExceptionBehavior();

  unsigned Opcode = getConstrainedOpcode(FPI.getIntrinsicID());
  if (!Opcode)
    return false;

  uint32_t Flags = MachineInstr::copyFlagsFromInstruction(FPI);
  if (EB == fp::ExceptionBehavior::ebIgnore)
    Flags |= MachineInstr::NoFPExcept;

  // G_STRICT_FCMP and G_STRICT_FCMPS carry a predicate operand (like G_FCMP)
  // and must be handled separately from the generic arithmetic path.
  if (Opcode == TargetOpcode::G_STRICT_FCMP ||
      Opcode == TargetOpcode::G_STRICT_FCMPS) {
    auto *FPCmp = cast<ConstrainedFPCmpIntrinsic>(&FPI);
    Register Op0 = getOrCreateVReg(*FPCmp->getArgOperand(0));
    Register Op1 = getOrCreateVReg(*FPCmp->getArgOperand(1));
    Register Res = getOrCreateVReg(FPI);
    MIRBuilder.buildInstr(Opcode, {Res}, {}, Flags)
        .addPredicate(FPCmp->getPredicate())
        .addUse(Op0)
        .addUse(Op1);
    return true;
  }

  SmallVector<llvm::SrcOp, 4> VRegs;
  for (unsigned I = 0, E = FPI.getNonMetadataArgCount(); I != E; ++I)
    VRegs.push_back(getOrCreateVReg(*FPI.getArgOperand(I)));

  MIRBuilder.buildInstr(Opcode, {getOrCreateVReg(FPI)}, VRegs, Flags);
  return true;
}
```

### GenericMachineInstrs.h

```cpp
/// Represent a G_STRICT_FCMP (quiet) or G_STRICT_FCMPS (signaling).
/// Operand layout: Res, Pred, LHS, RHS  -- same as G_FCMP.
class GStrictFCmp : public GenericMachineInstr {
public:
  CmpInst::Predicate getCond() const {
    return static_cast<CmpInst::Predicate>(getOperand(1).getPredicate());
  }
  Register getLHSReg() const { return getReg(2); }
  Register getRHSReg() const { return getReg(3); }
  bool isSignaling() const {
    return getOpcode() == TargetOpcode::G_STRICT_FCMPS;
  }

  static bool classof(const MachineInstr *MI) {
    return MI->getOpcode() == TargetOpcode::G_STRICT_FCMP ||
           MI->getOpcode() == TargetOpcode::G_STRICT_FCMPS;
  }
};
```
